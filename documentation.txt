--- Setup mail server ---
1.  sudo apt-get install mailutils
2.  sudo apt-get install postfix
3.  In /etc/postfix/main.cf find:
       
       mailbox_size_limit = 0
       recipient_delimiter = +
       inet_interfaces = all
       
       and replace "inet_interfaces = all" to "inet_interfaces = localhost"
4.  sudo echo "smtp_generic_maps = hash:/etc/postfix/generic" >> /etc/postfix/main.cf
5.  sudo echo 'bradensteffaniak@wolfpaq.co braden@wolfpaq.co' >> /etc/postfix/generic
6.  sudo postmap /etc/postfix/generic
7.  sudo echo "bradensteffaniak: braden@wolfpaq.co" >> /etc/aliases
8.  sudo newaliases
9.  Change /etc/mailname contents to wolfpaq.co
10. sudo service postfix restart
